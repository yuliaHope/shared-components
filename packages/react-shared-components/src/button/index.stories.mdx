### Button with a callback

Click on the button to trigger a callback function

<Preview>
  <Story name='WithOnclicHandler'>{() => {
    const clickHandler = (e) => {
      alert(`${e.currentTarget.dataset.type} clicked`);
    };
      const disabled = boolean('disabled', false);
      return (
          <div className="d-flex">
            <Button data-type="regular" className="btn-regular m-3" onClick={clickHandler} disabled={disabled}>
             Search
            </Button>
            <Button data-type="link" className="btn-link m-3" onClick={clickHandler} icon={IconCalendar} disabled={disabled} />
            <Button data-type="main" className="btn-main m-3" onClick={clickHandler} disabled={disabled}>
             Log in
           </Button>
        </div>
      )
  }}
  </Story>
</Preview>
